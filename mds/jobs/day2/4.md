4.  **테마 토글 컴포넌트 (`ThemeToggle.tsx`) 수정 (`remix-themes` 사용):**
    *   `app/components/ui/ThemeToggle.tsx` 파일을 생성하거나 (Shadcn add 시 생성됨) 기존 파일을 수정합니다.
    *   `next-themes` 대신 `remix-themes`의 `useTheme` 훅을 사용합니다.
    *   테마 변경 시 `@remix-run/react`의 `useFetcher`를 사용하여 `/action/set-theme` 라우트에 POST 요청을 보냅니다.

    ```bash
    # Shadcn UI로 dropdown-menu, button 컴포넌트가 추가되었다고 가정
    # 파일 경로는 components.json 설정에 따라 다를 수 있음 (예: ~/components/ui/ThemeToggle.tsx)
    touch app/components/ui/ThemeToggle.tsx 
    ```

    ```typescript
    // app/components/ui/ThemeToggle.tsx (수정됨)
    import * as React from "react";
    import { Moon, Sun } from "lucide-react";
    import { useFetcher } from "@remix-run/react"; // useFetcher import
    import { Theme, useTheme } from "remix-themes"; // remix-themes의 useTheme import

    import { Button } from "~/components/ui/button";
    import {
      DropdownMenu,
      DropdownMenuContent,
      DropdownMenuItem,
      DropdownMenuTrigger,
    } from "~/components/ui/dropdown-menu";

    export function ThemeToggle() {
      const [, setTheme] = useTheme(); // remix-themes의 useTheme 사용
      const fetcher = useFetcher(); // 테마 변경 action 호출용 fetcher

      // 테마 변경 함수
      const handleThemeChange = (theme: Theme | "system") => {
         // fetcher.Form을 사용하여 /action/set-theme으로 POST 요청
         fetcher.submit(
           { theme: theme }, 
           { method: "post", action: "/action/set-theme" }
         );
         // setTheme은 UI 즉시 반영용 (선택적)
         // setTheme(theme);
      };

      return (
        <DropdownMenu>
          <DropdownMenuTrigger asChild>
            <Button variant="outline" size="icon">
              <Sun className="h-[1.2rem] w-[1.2rem] rotate-0 scale-100 transition-all dark:-rotate-90 dark:scale-0" />
              <Moon className="absolute h-[1.2rem] w-[1.2rem] rotate-90 scale-0 transition-all dark:rotate-0 dark:scale-100" />
              <span className="sr-only">테마 변경</span>
            </Button>
          </DropdownMenuTrigger>
          <DropdownMenuContent align="end">
            <DropdownMenuItem onClick={() => handleThemeChange(Theme.LIGHT)}>
              라이트
            </DropdownMenuItem>
            <DropdownMenuItem onClick={() => handleThemeChange(Theme.DARK)}>
              다크
            </DropdownMenuItem>
            {/* remix-themes는 system 설정을 직접 지원하지 않으므로, 
                별도 로직 또는 OS 설정 감지 필요 시 추가 구현. 
                일단 라이트/다크만 제공 */}
            {/* <DropdownMenuItem onClick={() => handleThemeChange("system")}>
              시스템
            </DropdownMenuItem> */}
          </DropdownMenuContent>
        </DropdownMenu>
      );
    }
    ```

5.  **루트 레이아웃 (`app/root.tsx`)에 헤더/푸터 적용:**
    *   `App` 컴포넌트 내에서 `Header`와 `Footer`를 import하고 `Outlet` 주변에 배치합니다.

    ```typescript
    // app/root.tsx (App 컴포넌트 및 Provider 수정)
    import type { LinksFunction, LoaderFunctionArgs } from "@remix-run/node"; 
    import {
      Links,
      Meta,
      Outlet,
      Scripts,
      ScrollRestoration,
      useLoaderData,
      LiveReload, // LiveReload 추가 (개발 환경)
    } from "@remix-run/react";
    import { Header } from "./components/layout/Header"; // Header 임포트
    import { Footer } from "./components/layout/Footer"; // Footer 임포트
    import { themeSessionResolver } from "~/lib/theme.server"; // 테마 리졸버 임포트 경로 확인
    import {
      PreventFlashOnWrongTheme,
      ThemeProvider,
      useTheme,
    } from "remix-themes"; // remix-themes 관련 임포트
    import clsx from "clsx";
    import stylesheet from "~/tailwind.css?url"; // Tailwind CSS 임포트 확인

    // loader에서 테마 로드 (기존 로더와 병합)
    export const loader = async ({ request }: LoaderFunctionArgs) => {
      const { getTheme } = await themeSessionResolver(request);
      const theme = getTheme();
      // ... 기존 loader 로직 (env, session 등) ...
      // const env = { ... };
      // const { data: { session } } = ...;
      // return Response.json({ env, session, theme }, { headers: headers });
      return Response.json({ theme }); // 예시: 테마만 반환
    };

    // 링크 함수 (Tailwind CSS 포함)
    export const links: LinksFunction = () => [
      { rel: "stylesheet", href: stylesheet },
      // ... other links ...
    ];

    // 최상위 Provider 컴포넌트
    export default function AppWithProviders() {
      const { theme } = useLoaderData<typeof loader>();
      return (
        <ThemeProvider specifiedTheme={theme} themeAction="/action/set-theme">
          <App />
        </ThemeProvider>
      );
    }

    // 메인 앱 컴포넌트
    function App() {
      const data = useLoaderData<typeof loader>();
      const [theme] = useTheme(); // remix-themes 훅 사용

      return (
        <html lang="ko" className={clsx(theme)}>
          <head>
            <meta charSet="utf-8" />
            <meta name="viewport" content="width=device-width, initial-scale=1" />
            <Meta />
            <PreventFlashOnWrongTheme ssrTheme={Boolean(data.theme)} /> {/* FOUC 방지 */}
            <Links />
          </head>
          <body className="min-h-screen bg-background font-sans antialiased">
             <div className="relative flex min-h-screen flex-col"> {/* 전체 레이아웃 div */}
                <Header />
                <main className="flex-1"> {/* 메인 컨텐츠 영역 */}
                  <Outlet />
                </main>
                <Footer />
             </div>
            <ScrollRestoration />
            <Scripts />
            <LiveReload /> {/* 개발 환경 새로고침 */} 
          </body>
        </html>
      );
    }
    ```
